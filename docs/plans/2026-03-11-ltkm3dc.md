function _ltkm3dc_mean_positive_gap(values::AbstractVector{<:Real})
    sorted = sort!(collect(Float64.(values)))
    gaps = diff(sorted)
    positive = gaps[gaps .> 0]
    isempty(positive) && throw(ArgumentError("cannot estimate spacing from degenerate source coordinates"))
    return mean(positive)
end

function _ltkm3dc_estimate_kmax(sources::AbstractMatrix{<:Real})
    src = as_3xn(sources)
    dx = _ltkm3dc_mean_positive_gap(view(src, 1, :))
    dy = _ltkm3dc_mean_positive_gap(view(src, 2, :))
    dz = _ltkm3dc_mean_positive_gap(view(src, 3, :))
    return min(pi / dx, pi / dy, pi / dz)
end
